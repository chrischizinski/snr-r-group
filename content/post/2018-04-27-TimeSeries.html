Plot mentions of drought in the news

spop %>% 
  mutate(valDate = ymd(ReleaseDate),
         wsum = D0 + D1 + D2 + D3 + D4,
         year = year(valDate),
         month = month(valDate),
         yrmo = paste0(year,"",month),
         yrmo = parse_date_time(yrmo, "ym"),
         yrmo = as.Date(yrmo)) -> spopmo
