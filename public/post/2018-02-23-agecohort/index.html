Filter the data

 huntData %>% 
  filter(gender == "Male",
         residency == "resident",
         permitType %in% c("Hunt Fish Combo","Hunt"),
         permitYear %in% c(2008:2016)) -> permits_oi

popData %>% 
  filter(is.na(Notes),
         Gender == "Male",
         Year %in% c(2008:2016)) %>% 
  select(`Age Code`, Year, `Projected Populations`) %>% 
  rename(Age = `Age Code`,
         Pop = `Projected Populations`) -> pop_oi
